    function processViewAsSelection(cuid, origCuid) {
        var form      = document.ViewAsSelection;
        var vadd      = form.ViewAsDropdown;
        var vaddIndex = vadd.selectedIndex;
        var vaddValue = vadd.options[vaddIndex].value;
        var delimIdx  = vaddValue.indexOf('%');
        var rform     = document.RvpSelection;
        var rddValue  = "";
        if (rform != null) {
            var rdd      = rform.RvpDropdown;
            var rddIndex = rdd.selectedIndex;
            rddValue     = rdd.options[rddIndex].value;
        }
        if (delimIdx != -1) {
            var salesIds        = vaddValue.substring(0, delimIdx);
            var role            = vaddValue.substring(delimIdx+1, vaddValue.length);
            form.salesIds.value = salesIds;
            form.role.value     = role;
            form.origCuid.value  = origCuid;
            form.cuid.value     = cuid;
            form.rv.value       = rddValue;
            form.sv.value       = vaddValue;
            form.submit();
        }
    }
